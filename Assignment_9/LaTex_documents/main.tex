\section{Question}
\begin{frame}{Question}
\textbf{Papoulli Chapter 9(Ex 9.43)} : 
The process x(t) is WSS and $R_{xx}(\tau) = 5\delta(\tau)$\\
(a) Find $E\{y^2(t)\}$ and $S_{yy}(\omega)$ if y'(t) +3y(t) = x(t)\\
(b) Find $E\{y^2(t)\}$ and $R_{xy}(t_1, t_2)$ if y'(t) + 3y(t) =x(t)U(t).\\

\end{frame}
\section{Theory}
  \begin{frame}{Theory Page 1}
If h(t) is of short duration relative to the variations of q(t) ,then
\begin{align}
    E\{y^2(t)\} \approx q(t)\int_{-\infty}^{\infty}|h(\alpha)|^2d\alpha = Eq(t)
\end{align}
This relationship justifies the term average intensity used to describe the function q(t)\\
  \end{frame}
  \begin{frame}{Theory Page 2}
      If a stationary process v(t) with autocorrelation $R_{vv}(\tau) = q\delta(t)$ is applied at t=0 to a linear system with 
      \begin{align*}
          h(t) = e^{-ct}U(t)
      \end{align*}
      The autocorrelation of the resulting output y(t) equals 
      \begin{align}
          E\{y^2(t)\} = R_{yy}(t, t) = \frac{q}{2c}(1 - e^{-2ct}) = q\int_{0}^{t}h^2(\alpha)d\alpha
      \end{align}
  \end{frame}
\section{Solution}
\begin{frame}{Solution Page 1}
(a) y'(t) + 3y(t) = x(t), $R_{XX}(\tau) = 5\delta(\tau)$. The process y(t) is the output of the system 
\begin{align}
    H(s) = \frac{1}{s+3}\\
    h(t) = e^{-3t}U(t)
\end{align}
Hence from (1)
\begin{align}
    E\{y^2(t)\} = 5\int_{0}^{\infty}e^{-6t}dt = \frac{5}{6}\\
    S_{yy}(\omega) = \frac{5}{\omega^2 + 9} \;\;\;\;\;\;\;\;\;\;\; R_{yy}(\tau) = \frac{5}{6}e^{-3|\tau|}
\end{align}
\end{frame}
 \begin{frame}{Solution Page 2}
(b) From (2) 
\begin{align}
    E\{y^2(t)\} &= 5\int_{0}^{t}e^{-6\alpha}d\alpha = \frac{5}{6}(1 - e^{-6t}) \;\;\;\;\;\;\;\;\;\;\;\;\; t >0\\
    R_{xy}(t_1, t_2) &= 5e^{-2|t_2 - t_1|}U(t_1)U(t_2)U(t_2 - t_1)
\end{align}
 \end{frame}
